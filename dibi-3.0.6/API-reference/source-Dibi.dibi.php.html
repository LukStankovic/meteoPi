121:      * @param  string   connection registy name
122:      * @return Dibi\Connection
123:      * @throws Dibi\Exception
124:      */
125:     public static function getConnection($name = NULL)
126:     {
127:         if ($name === NULL) {
128:             if (self::$connection === NULL) {
129:                 throw new Dibi\Exception('Dibi is not connected to database.');
130:             }
131: 
132:             return self::$connection;
133:         }
134: 
135:         if (!isset(self::$registry[$name])) {
136:             throw new Dibi\Exception("There is no connection named '$name'.");
137:         }
138: 
139:         return self::$registry[$name];
140:     }
