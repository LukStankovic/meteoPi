35:      * After event notification.
36:      * @return void

38:     public function logEvent(Dibi\Event $event)
39:     {
40:         if (($event->type & $this->filter) === 0) {
41:             return;
42:         }
43: 
44:         $handle = fopen($this->file, 'a');
45:         if (!$handle) {
46:             return; // or throw exception?
47:         }
48:         flock($handle, LOCK_EX);
49: 
50:         if ($event->result instanceof \Exception) {
51:             $message = $event->result->getMessage();
52:             if ($code = $event->result->getCode()) {
53:                 $message = "[$code] $message";
54:             }
55:             fwrite($handle,
56:                 "ERROR: $message"
57:                 . "\n-- SQL: " . $event->sql
58:                 . "\n-- driver: " . $event->connection->getConfig('driver') . '/' . $event->connection->getConfig('name')
59:                 . ";\n-- " . date('Y-m-d H:i:s')
60:                 . "\n\n"
61:             );
62:         } else {
63:             fwrite($handle,
64:                 'OK: ' . $event->sql
65:                 . ($event->count ? ";\n-- rows: " . $event->count : '')
66:                 . "\n-- takes: " . sprintf('%0.3f ms', $event->time * 1000)
67:                 . "\n-- source: " . implode(':', $event->source)
68:                 . "\n-- driver: " . $event->connection->getConfig('driver') . '/' . $event->connection->getConfig('name')
69:                 . "\n-- " . date('Y-m-d H:i:s')
70:                 . "\n\n"
71:             );
72:         }
73:         fclose($handle);
74:     }
