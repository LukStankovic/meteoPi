 54:      * Connects to a database.
 55:      * @return void
 56:      * @throws Dibi\Exception
 57:      */
 58:     public function connect(array & $config)
 59:     {
 60:         $error = NULL;
 61:         if (isset($config['resource'])) {
 62:             $this->connection = $config['resource'];
 63: 
 64:         } else {
 65:             $config += [
 66:                 'charset' => 'utf8',
 67:             ];
 68:             if (isset($config['string'])) {
 69:                 $string = $config['string'];
 70:             } else {
 71:                 $string = '';
 72:                 Dibi\Helpers::alias($config, 'user', 'username');
 73:                 Dibi\Helpers::alias($config, 'dbname', 'database');
 74:                 foreach (['host', 'hostaddr', 'port', 'dbname', 'user', 'password', 'connect_timeout', 'options', 'sslmode', 'service'] as $key) {
 75:                     if (isset($config[$key])) {
 76:                         $string .= $key . '=' . $config[$key] . ' ';
 77:                     }
 78:                 }
 79:             }
 80: 
 81:             set_error_handler(function($severity, $message) use (& $error) {
 82:                 $error = $message;
 83:             });
 84:             if (empty($config['persistent'])) {
 85:                 $this->connection = pg_connect($string, PGSQL_CONNECT_FORCE_NEW);
 86:             } else {
 87:                 $this->connection = pg_pconnect($string, PGSQL_CONNECT_FORCE_NEW);
 88:             }
 89:             restore_error_handler();
 90:         }
 91: 
 92:         if (!is_resource($this->connection)) {
 93:             throw new Dibi\DriverException($error ?: 'Connecting error.');
 94:         }
 95: 
 96:         pg_set_error_verbosity($this->connection, PGSQL_ERRORS_VERBOSE);
 97: 
 98:         if (isset($config['charset']) && pg_set_client_encoding($this->connection, $config['charset'])) {
 99:             throw self::createException(pg_last_error($this->connection));
100:         }
101: 
102:         if (isset($config['schema'])) {
103:             $this->query('SET search_path TO "' . $config['schema'] . '"');
104:         }
105:     }

134:     public function query($sql)
135:     {

137:         $res = @pg_query($this->connection, $sql); // intentionally @

151:     /**
152:      * @return Dibi\DriverException
153:      */
154:     public static function createException($message, $code = NULL, $sql = NULL)
155:     {
156:         if ($code === NULL && preg_match('#^ERROR:\s+(\S+):\s*#', $message, $m)) {
157:             $code = $m[1];
158:             $message = substr($message, strlen($m[0]));
159:         }
160: 
161:         if ($code === '0A000' && strpos($message, 'truncate') !== FALSE) {
162:             return new Dibi\ForeignKeyConstraintViolationException($message, $code, $sql);
163: 
164:         } elseif ($code === '23502') {
165:             return new Dibi\NotNullConstraintViolationException($message, $code, $sql);
166: 
167:         } elseif ($code === '23503') {
168:             return new Dibi\ForeignKeyConstraintViolationException($message, $code, $sql);
169: 
170:         } elseif ($code === '23505') {
171:             return new Dibi\UniqueConstraintViolationException($message, $code, $sql);
172: 
173:         } else {
174:             return new Dibi\DriverException($message, $code, $sql);
175:         }
176:     }

531:      * Returns metadata for all columns in a table.
532:      * @param  string
533:      * @return array
534:      */
535:     public function getColumns($table)
536:     {
537:         $_table = $this->escapeText($this->escapeIdentifier($table));
538:         $res = $this->query("
539:             SELECT indkey
540:             FROM pg_class
541:             LEFT JOIN pg_index on pg_class.oid = pg_index.indrelid AND pg_index.indisprimary
542:             WHERE pg_class.oid = $_table::regclass
543:         ");
544:         $primary = (int) pg_fetch_object($res->resultSet)->indkey;
545: 
546:         $res = $this->query("
547:             SELECT *
548:             FROM information_schema.columns c
549:             JOIN pg_class ON pg_class.relname = c.table_name
550:             JOIN pg_namespace nsp ON nsp.oid = pg_class.relnamespace AND nsp.nspname = c.table_schema
551:             WHERE pg_class.oid = $_table::regclass
552:             ORDER BY c.ordinal_position
553:         ");
554: 
555:         if (!$res->getRowCount()) {
556:             $res = $this->query("
557:                 SELECT
558:                     a.attname AS column_name,
559:                     pg_type.typname AS udt_name,
560:                     a.attlen AS numeric_precision,
561:                     a.atttypmod-4 AS character_maximum_length,
562:                     NOT a.attnotnull AS is_nullable,
563:                     a.attnum AS ordinal_position,
564:                     adef.adsrc AS column_default
565:                 FROM
566:                     pg_attribute a
567:                     JOIN pg_type ON a.atttypid = pg_type.oid
568:                     JOIN pg_class cls ON a.attrelid = cls.oid
569:                     LEFT JOIN pg_attrdef adef ON adef.adnum = a.attnum AND adef.adrelid = a.attrelid
570:                 WHERE
571:                     cls.relkind IN ('r', 'v', 'mv')
572:                     AND a.attrelid = $_table::regclass
573:                     AND a.attnum > 0
574:                     AND NOT a.attisdropped
575:                 ORDER BY ordinal_position
576:             ");
577:         }
578: 
579:         $columns = [];
580:         while ($row = $res->fetch(TRUE)) {
581:             $size = (int) max($row['character_maximum_length'], $row['numeric_precision']);
582:             $columns[] = [
583:                 'name' => $row['column_name'],
584:                 'table' => $table,
585:                 'nativetype' => strtoupper($row['udt_name']),
586:                 'size' => $size > 0 ? $size : NULL,
587:                 'nullable' => $row['is_nullable'] === 'YES' || $row['is_nullable'] === 't',
